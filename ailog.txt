class TmdbParser:
"""TMDB API 응답 → 내부 도메인 모델 변환 전담"""

def parse_cast(self, detail):
    return [
        {
            "name":      c.get("name", ""),
            "character": c.get("character", ""),
            "photo":     self._img_url(c.get("profile_path"), "w185"),
        }
        for c in (detail.get("credits") or {}).get("cast", [])[:10]
    ]

def parse_reviews(self, detail):
    return [
        {
            "author":  r.get("author", "익명"),
            "content": r.get("content", "")[:400],
            "rating":  (r.get("author_details") or {}).get("rating"),
        }
        for r in (detail.get("reviews") or {}).get("results", [])[:5]
    ]

def parse_genres(self, detail):
    return [g["name"] for g in (detail.get("genres") or [])]

def _img_url(self, path, size="w500"):
    return f"https://image.tmdb.org/t/p/{size}{path}" if path else None

class ContentFacade:
"""
외부에서 보기엔 단순한 인터페이스.

self._ott = OttService()

def get_detail(self, media_type, content_id):
    raw = tmdb_get(
        f"/{media_type}/{content_id}",
        {"append_to_response": "credits,reviews"}
    )
    if not raw:
        return None

    return {
        "id":           content_id,
        "media_type":   media_type,
        "title":        raw.get("title") or raw.get("name"),
        "overview":     raw.get("overview", ""),
        "poster":       self._parser.img_url(raw.get("poster_path")),
        "backdrop":     self._parser.img_url(raw.get("backdrop_path"), "w1280"),
        "rating":       round(float(raw.get("vote_average") or 0), 1),
        "genres":       self._parser.parse_genres(raw),
        "cast":         self._parser.parse_cast(raw),
        "tmdb_reviews": self._parser.parse_reviews(raw),
        "providers":    self._ott.get_providers(media_type, content_id),
    }

from abc import ABC, abstractmethod

class MediaTypeHandler(ABC):
@AbstractMethod
def discover_path(self) -> str: ...

@abstractmethod
def genre_path(self) -> str: ...

def is_valid(self) -> bool:
    return True
class MovieHandler(MediaTypeHandler):
def discover_path(self): return "/discover/movie"
def genre_path(self): return "/genre/movie/list"

class TvHandler(MediaTypeHandler):
def discover_path(self): return "/discover/tv"
def genre_path(self): return "/genre/tv/list"

HANDLERS: dict[str, MediaTypeHandler] = {
"movie": MovieHandler(),
"tv": TvHandler(),
}

def get_handler(media_type: str) -> MediaTypeHandler:
handler = HANDLERS.get(media_type)
if not handler:
raise ValueError(f"지원하지 않는 media_type: '{media_type}'")
return handler

from werkzeug.routing import BaseConverter, ValidationError

VALID_MEDIA_TYPES = {"movie", "tv"}

class MediaTypeConverter(BaseConverter):
def to_python(self, value: str) -> str:
if value not in VALID_MEDIA_TYPES:
raise ValidationError(f"지원하지 않는 media_type: {value}")
return value

def to_url(self, value: str) -> str:
    return value

Decision & Review Process ->

[전략 3] Parser Class 도입 (SRP 위반 해결)
변경의 국소화: 데이터 가공 규칙(글자 수 제한, 이미지 URL 조립 등)이 바뀌어도 라우트 함수는 건드릴 필요 없이 파서만 수정하면 되도록 격리함.
parser class 도입을 통해 feature envy 문제까지 해결할 수 있음
테스트 독립성: 실제 API 호출 없이도 특정 JSON 데이터를 넣었을 때 가공 로직이 의도대로 작동하는지 단위 테스트(Unit Test)를 수행하기 위함임.

[전략 5] Repository Pattern 도입 (DRY 위반 해결)
중복 제거(DRY): movie와 tv 목록 호출 시 공통으로 사용하는 파라미터 조립 로직을 중앙화하여 코드 양을 줄이고 복사·붙여넣기로 인한 실수를 방지함.
데이터 소스 추상화: 라우트 함수가 TMDB API의 구체적인 쿼리 구조를 알 필요 없게 만들어, 나중에 데이터베이스나 캐싱 레이어로 교체하더라도 비즈니스 로직을 보호하기 위함임.

[전략 4] Strategy Pattern 도입 (OCP 위반 해결)
확장성(OCP): 새로운 미디어 타입(애니메이션 등) 추가 시 기존의 if/elif 조건문을 수정하는 대신, 새 클래스만 추가하여 등록하면 되는 유연한 구조를 확보함.
방어적 설계: 대소문자나 공백 등 사용자 입력 오류를 핸들러 획득 단계에서 원천 차단하고, 내부 레지스트리를 외부 조작으로부터 보호(Encapsulation)하여 시스템 안정성을 높이기 위함임.

[전략 4] Repository Pattern 도입

데이터 캡슐화: media_type과 content_id를 조합해 키(Key)를 생성하는 상세 로직을 외부 라우트 함수가 알 필요 없도록 저장소 내부로 숨김 처리함.

유지보수 효율성: 리뷰를 저장하는 방식이 단순 딕셔너리에서 Redis나 DB로 변경되더라도, 라우트 코드 수정 없이 레포지토리 클래스만 교체하면 되는 유연성을 확보함.

도메인 로직 집중: "키를 어떻게 만들 것인가"라는 파편화된 고민을 레포지토리 한 곳으로 몰아넣어, 서비스 전체의 데이터 관리 일관성을 높이기 위함임.
